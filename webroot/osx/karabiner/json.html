karabiner.json Reference Manual

File location

~/.config/karabiner/karabiner.json Main configuration file.
Karabiner-Elements watches this file and reload it automatically when it is updated.
~/.config/karabiner/assets/complex_modifications Imported complex_modifications files.
You can use them in Preferences > Complex Modifications > Add rule.

karabiner.json syntax

The basic syntax of karabiner.json

{
    "global": {
        ...
    },
    "profiles": [
        {
            "name": "Profile name",
            "selected": true,
            "simple_modifications": [
                ...
            ],
            "fn_function_keys": [
                ...
            ],
            "complex_modifications": {
                "parameters": {
                    ...
                },
                "rules": [
                    ...
                ]
            },
            "virtual_hid_keyboard": {
                "keyboard_type": "ansi",
                "caps_lock_delay_milliseconds": 0
            },
            "devices": [
                ...
            ]
        },
        {
            "name": "Profile name",
            "selected": false,
            ...
        },
        ...
    ]
}

The syntax of complex_modifications rules

{
    "complex_modifications": {
        "parameters": {
            ...
        },
        "rules": [
            {
                "description": "description of the rule",
                "manipulators": [
                    {
                        "type": "basic",
                        "from": {
                            "key_code": "key_code name which you want to change",
                            "modifiers": {
                                "mandatory": [
                                    "modifier name",
                                    ...
                                ],
                                "optional": [
                                    "modifier name",
                                    ...
                                ]
                            }
                        },
                        "to": [
                            {
                                "key_code": "key_code name",
                                "modifiers": [
                                    "modifier name",
                                    ...
                                ]
                            }
                        ]
                    },
                    {
                        "type": "basic",
                        ...
                    },
                    ...
                ]
            },
            {
                "description": "description of the rule",
                "manipulators": [
                    ...
                ]
            },
            ...
        ]
    }
}
